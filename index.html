function generateBoard() { // function ຊ່ວຍສຸ່ມເລກ 5 ໂຕບໍ່ຊ້ຳກັນໃນຊ່ວງທີ່ກຳນົດ ແລະ ຈັດລຽງຈາກນ້ອຍຫາໃຫຍ່ const getCol = (min, max) => { let nums = []; while(nums.length < 5) { let r = Math.floor(Math.random() * (max - min + 1)) + min; if(!nums.includes(r)) nums.push(r); } return nums.sort((a, b) => a - b); // ລຽງຈາກນ້ອຍຫາໃຫຍ່ }; // ສ້າງແຕ່ລະຖັນ let b = getCol(1, 15); let i = getCol(16, 30); let n = getCol(31, 45); let g = getCol(46, 60); let o = getCol(61, 75); let boardArr = []; // ເອົາເລກມາລວມກັນແບບແຖວ (Row by Row) ຕາມຮູບແບບ Bingo 5x5 for(let row = 0; row < 5; row++) { boardArr.push(b[row], i[row], n[row], g[row], o[row]); } // ກຳນົດຕາຕະລາງກາງໃຫ້ເປັນ FREE boardArr[12] = "FREE"; const board = document.getElementById('bingo-board'); board.innerHTML = ""; boardArr.forEach((val) => { const div = document.createElement('div'); div.className = "cell" + (val === "FREE" ? " free selected" : ""); div.innerText = val === "FREE" ? "★" : val; div.onclick = () => { const called = document.getElementById('called-numbers').innerText.split(', '); // ຖ້າເປັນຊ່ອງ FREE ຫຼື ເລກນັ້ນຖືກຂານແລ້ວ ໃຫ້ປ່ຽນສີ if (val === "FREE" || called.includes(val.toString())) { div.classList.add('selected'); if (checkWin()) update(gameRef, { winner: pName }); } }; board.appendChild(div); }); }
